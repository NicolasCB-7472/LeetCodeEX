class Solution {
    public void duplicateZeros(int[] arr) {
        int l=arr.length;
        ArrayList<Integer> ar=new ArrayList<>();
        for(int i:arr){
            if(i==0) ar.add(0);
            ar.add(i);
        }
        for(int i=0;i<l;i++){
            arr[i]=ar.get(i);
        }
    }
}
